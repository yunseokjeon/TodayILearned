40. Combination Sum II

https://leetcode.com/problems/combination-sum-ii/

```Java
class Solution {
    public List<List<Integer>> combinationSum2(int[] candidates, int target) {
        Arrays.sort(candidates);
        LinkedList<Integer> combination = new LinkedList<>();
        List<List<Integer>> results = new ArrayList<>();
        backtrack(candidates, target, 0, combination, results);
        return results;
    }

    public void backtrack(int[] candidates, int remain, int current,
                          LinkedList<Integer> combination, List<List<Integer>> results) {
        if (remain == 0) {
            results.add(new ArrayList<>(combination));
            return;
        }
        for (int next = current; next < candidates.length; next++) {
            if (next > current && candidates[next - 1] == candidates[next]) {
                continue;
            }
            Integer pick = candidates[next];
            if (remain - pick < 0) {
                break;
            }
            combination.add(pick);
            backtrack(candidates, remain - pick, next + 1, combination, results);
            combination.removeLast();
        }
    }
}
```
