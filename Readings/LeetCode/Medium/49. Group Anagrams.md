49. Group Anagrams

https://leetcode.com/problems/group-anagrams/

```TypeScript
function groupAnagrams(strs: string[]): string[][] {
    let strMap = {};
    for (let str of strs) {
        let key = str.split("").sort().join("");
        if (strMap[key]) {
            strMap[key].push(str);
        } else {
            strMap[key] = [str];
        }
    }
    return Object.values(strMap);
};
```
