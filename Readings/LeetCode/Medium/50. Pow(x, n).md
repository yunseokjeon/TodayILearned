50. Pow(x, n)

https://leetcode.com/problems/powx-n/

```TypeScript
function myPow(x: number, n: number): number {
    if (n === 0) {
        return 1;
    }
    if (n < 0) {
        return myPow(1 / x, -n);
    }
    let candidate: number = myPow(x, Math.floor(n / 2));
    if (n % 2 === 0) {
        return candidate * candidate;
    }
    return x * candidate * candidate;
};
```
