1885. Count Pairs in Two Arrays

https://leetcode.com/problems/count-pairs-in-two-arrays/

```Java
class Solution {
    public long countPairs(int[] nums1, int[] nums2) {
        int N = nums1.length;
        int[] diff = new int[N];
        for (int i = 0; i < N; i++) {
            diff[i] = nums1[i] - nums2[i];
        }
        int left = 0, right = N - 1;
        long result = 0;
        Arrays.sort(diff);
        while (left < right) {
            if (diff[left] + diff[right] > 0) {
                result += (right - left);
                right--;
            } else {
                left++;
            }
        }
        return result;
    }
}
```
